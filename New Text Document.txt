2.	Popups “af:popup” should hold directly a dialog “af:dialog”, i.e. in the first level. Dialog’s “modal” property default values is “true”, this means that when the dialog or popup is visible, it blocks user input to all other windows in the program.

// Binary Search

    static int  binarySearch(Integer arr[], int l, int r, int x) 
    { 
        int mid = 0;
        if (r >= l) { 
            mid = l + (r - l) / 2; 
  
            // If the element is present at the 
            // middle itself 
            if (arr[mid] == x) 
                return mid; 
  
            // If element is smaller than mid, then 
            // it can only be present in left sub array 
            if (arr[mid] > x) 
                return binarySearch(arr, l, mid - 1, x); 
  
            // Else the element can only be present 
            // in right sub array 
            return binarySearch(arr, mid + 1, r, x); 
        } 
        
  
        // We reach here when element is not present 
        // in array 

        return -1; 
    }
